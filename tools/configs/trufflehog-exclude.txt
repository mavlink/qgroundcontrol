# TruffleHog exclusion patterns (regex format)
# https://trufflesecurity.com/trufflehog

# Test fixture data (not test code - secrets in test code should be caught)

.*/fixtures/.*

.*/vendor/.*
.*/node_modules/.*
.*/\.git/.*

# Lock files and minified assets
.*\.lock$
.*package-lock\.json$
.*\.min\.js$
.*\.min\.css$

# Build artifacts
.*/build/.*
.*/dist/.*
